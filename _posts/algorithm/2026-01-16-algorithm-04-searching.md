```python
def interpolation_search(arr, target):
    low = 0
    high = len(arr) - 1

    while low <= high and arr[low] <= target <= arr[high]:
        if low == high:
            if arr[low] == target:
                return low
            return -1

        # 媛믪쓽 遺꾪룷瑜?怨좊젮???꾩튂 怨꾩궛
        pos = low + int(((target - arr[low]) * (high - low)) / (arr[high] - arr[low]))

            return pos
        elif arr[pos] < target:
            low = pos + 1
        else:
            high = pos - 1

    return -1
```
